# -----------------------------
# How to start the shimmer on linux
# -----------------------------

#turn on the shimmer, leave it on the dock station and slide up the power on button. If lights are blue and greeen, one still and one blinking it means the shimmer is looking for devices.

#run `conda deactivate` if the env is on. Then:
	cd pyshimmer/
	source Shimenv/bin/activate

# Open bluetoothctl
	bluetoothctl

# Inside bluetoothctl:
	scan on
# Note your Shimmer MAC address (format: XX:XX:XX:XX:XX:XX)  00:06:66:D7:C9:C7	
pair XX:XX:XX:XX:XX:XX		PIN code:1234
	trust XX:XX:XX:XX:XX:XX
	exit

# Bind the device to a serial port
	sudo rfcomm bind /dev/rfcomm0 XX:XX:XX:XX:XX:XX 1
	sudo rfcomm bind /dev/rfcomm0 00:06:66:D7:C9:C7 1

# Check that the port exists
	ls /dev/rfcomm*
